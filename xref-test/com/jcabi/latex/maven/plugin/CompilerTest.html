20  /**
21   * Test case for {@link Compiler}.
22   *
23   * @since 1.0
24   */
25  final class CompilerTest {
26  
27      @Test
28      @Timeout(120)
29      void compilesSimpleLatexSourceIntoPng(@TempDir final Path dir)
30          throws Exception {
31          Assumptions.assumeTrue(
32              CompilerTest.hasBinary("pdflatex")
33                  && CompilerTest.hasBinary("gs")
34                  && CompilerTest.hasBinary("pnmtopng"),
35              "pdflatex, ghostscript or netpbm not installed, skipping"
36          );
37          final File sources = dir.resolve("src").toFile();
38          sources.mkdirs();
39          Files.writeString(
40              sources.toPath().resolve("hello.tex"),
41              "\\documentclass{article}\\begin{document}Hi\\end{document}",
42              StandardCharsets.UTF_8
43          );
44          final Output output = new Compiler(
45              dir.resolve("temp").toFile()
46          ).compile(
47              new Source(sources, "hello.tex", Collections.emptyList())
48          );
49          final File save = dir.resolve("out").toFile();
50          save.mkdirs();
51          output.saveTo(save);
52          MatcherAssert.assertThat(
53              "PNG file is missing after compiling a simple LaTeX source",
54              new File(save, "hello.png").exists(),
55              Matchers.is(true)
56          );
57      }
58  
59      private static boolean hasBinary(final String name) {
60          final String osname = System.getProperty("os.name");
61          final String which;
62          if (osname != null && osname.startsWith("Windows")) {
63              which = "where";
64          } else {
65              which = "which";
66          }
67          boolean found;
68          try {
69              found = new ProcessBuilder(which, name).start().waitFor() == 0;
70          } catch (final IOException | InterruptedException ex) {
71              Thread.currentThread().interrupt();
72              found = false;
73          }
74          return found;
75      }
76  }
